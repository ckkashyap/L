**Goal:** Implement a complete vi editor (normal/insert/visual modes, macros, :commands, search) in pure L with minimal C FFI, running in both terminal and graphical (SDL) modes.

**Architecture:** Coroutine pipeline `key-source → macro-filter → mode-dispatch → renderer`. Platform seams (key-source, renderer) differ between terminal/SDL; all editor logic is shared pure L in `lib/vi/`.

**Tech Stack:** L interpreter (pvec from lib/ds.l, coroutines, strings, plists), `src/term_helper.c` (~70 lines C for raw mode/terminal size/timed reads), ANSI escapes for terminal rendering, existing `sdl_gfx.c` + new `sdl-wait-event` for SDL mode.

```c
/* src/term_helper.c — terminal FFI helper for vi editor */
#ifdef _WIN32
#  define WIN32_LEAN_AND_MEAN
#  include <windows.h>
#  define EXPORT __declspec(dllexport)
#else
#  include <termios.h>
#  include <unistd.h>
#  include <sys/ioctl.h>
#  include <sys/select.h>
#  define EXPORT
#endif
#include <stdio.h>

#ifdef _WIN32
static HANDLE g_hin = INVALID_HANDLE_VALUE;
static DWORD  g_orig_mode = 0;
#else
static struct termios g_orig;
#endif

EXPORT int term_set_raw(int on) {
#ifdef _WIN32
    if (g_hin == INVALID_HANDLE_VALUE)
        g_hin = GetStdHandle(STD_INPUT_HANDLE);
    if (on) {
        GetConsoleMode(g_hin, &g_orig_mode);
        DWORD m = g_orig_mode & ~(ENABLE_LINE_INPUT|ENABLE_ECHO_INPUT|ENABLE_PROCESSED_INPUT);
        return SetConsoleMode(g_hin, m) ? 0 : -1;
    }
    return SetConsoleMode(g_hin, g_orig_mode) ? 0 : -1;
#else
    if (on) {
        struct termios raw;
        if (tcgetattr(STDIN_FILENO, &g_orig) < 0) return -1;
        raw = g_orig;
        cfmakeraw(&raw);
        raw.c_cc[VMIN] = 1; raw.c_cc[VTIME] = 0;
        return tcsetattr(STDIN_FILENO, TCSAFLUSH, &raw);
    }
    return tcsetattr(STDIN_FILENO, TCSAFLUSH, &g_orig);
#endif
}

EXPORT int term_get_cols(void) {
#ifdef _WIN32
    CONSOLE_SCREEN_BUFFER_INFO csbi;
    if (!GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi)) return 80;
    return csbi.srWindow.Right - csbi.srWindow.Left + 1;
#else
    struct winsize ws;
    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) < 0 || ws.ws_col == 0) return 80;
    return ws.ws_col;
#endif
}

EXPORT int term_get_rows(void) {
#ifdef _WIN32
    CONSOLE_SCREEN_BUFFER_INFO csbi;
    if (!GetConsoleScreenBufferInfo(GetStdHandle(STD_OUTPUT_HANDLE), &csbi)) return 24;
    return csbi.srWindow.Bottom - csbi.srWindow.Top + 1;
#else
    struct winsize ws;
    if (ioctl(STDOUT_FILENO, TIOCGWINSZ, &ws) < 0 || ws.ws_row == 0) return 24;
    return ws.ws_row;
#endif
}

EXPORT int term_read_byte(void) {
#ifdef _WIN32
    if (g_hin == INVALID_HANDLE_VALUE)
        g_hin = GetStdHandle(STD_INPUT_HANDLE);
    INPUT_RECORD ir; DWORD n;
    while (1) {
        if (!ReadConsoleInputA(g_hin, &ir, 1, &n)) return -1;
        if (ir.EventType == KEY_EVENT && ir.Event.KeyEvent.bKeyDown) {
            char c = ir.Event.KeyEvent.uChar.AsciiChar;
            if (c) return (unsigned char)c;
        }
    }
#else
    unsigned char b;
    return (read(STDIN_FILENO, &b, 1) == 1) ? (int)b : -1;
#endif
}

EXPORT int term_read_byte_timeout(int ms) {
#ifdef _WIN32
    if (g_hin == INVALID_HANDLE_VALUE)
        g_hin = GetStdHandle(STD_INPUT_HANDLE);
    if (WaitForSingleObject(g_hin, (DWORD)ms) != WAIT_OBJECT_0) return -1;
    return term_read_byte();
#else
    fd_set fds; FD_ZERO(&fds); FD_SET(STDIN_FILENO, &fds);
    struct timeval tv = { ms/1000, (ms%1000)*1000 };
    if (select(STDIN_FILENO+1, &fds, NULL, NULL, &tv) <= 0) return -1;
    unsigned char b;
    return (read(STDIN_FILENO, &b, 1) == 1) ? (int)b : -1;
#endif
}
```

```lisp
# lib/vi/buffer.l — pvec-of-strings buffer
# Requires lib/ds.l to be loaded first (pvec, pvec-push, pvec-get, pvec-put, pvec-count, pvec-seq)

(de buf-from-lines (lst)
  (let ((b (pvec)))
    (let loop ((b b) (l lst))
      (if l (loop (pvec-push b (car l)) (cdr l)) b))))

(de buf-empty ()       (buf-from-lines '("")))
(de buf-nlines (buf)   (pvec-count buf))
(de buf-line (buf row) (pvec-get buf (1- row)))

(de buf-set-line (buf row str)
  (pvec-put buf (1- row) str))

(de buf-ins-line (buf row str)
  # Insert str before row (1-based). Rows >= row shift down by 1.
  (let* ((lst (pvec-seq buf))
         (idx (1- row))
         (pre (let loop ((l lst) (i 0) (acc NIL))
                (if (or (null l) (= i idx)) (reverse acc)
                  (loop (cdr l) (1+ i) (cons (car l) acc)))))
         (post (let loop ((l lst) (i 0))
                 (if (= i idx) l (loop (cdr l) (1+ i))))))
    (buf-from-lines (append pre (cons str post)))))

(de buf-del-line (buf row)
  (let* ((lst (pvec-seq buf))
         (idx (1- row))
         (new (let loop ((l lst) (i 0) (acc NIL))
                (if (null l) (reverse acc)
                  (loop (cdr l) (1+ i)
                    (if (= i idx) acc (cons (car l) acc)))))))
    (buf-from-lines (if new new '("")))))

(de buf-line-len (buf row)
  (length (unpack (buf-line buf row))))

(de buf-ins-char (buf row col cp)
  (let ((line (buf-line buf row)))
    (buf-set-line buf row
      (pack (list (sub line 0 col) (char cp) (sub line col))))))

(de buf-del-char (buf row col)
  (let ((line (buf-line buf row)))
    (buf-set-line buf row
      (pack (list (sub line 0 col) (sub line (1+ col)))))))

(de buf-split-line (buf row col)
  (let ((line (buf-line buf row)))
    (buf-ins-line (buf-set-line buf row (sub line 0 col))
                  (1+ row) (sub line col))))

(de buf-join-lines (buf row)
  (let ((a (buf-line buf row))
        (b (buf-line buf (1+ row))))
    (buf-del-line (buf-set-line buf row (pack (list a b))) (1+ row))))

(de buf-to-string (buf)
  (let* ((lines (pvec-seq buf))
         (r (car lines))
         (rest (cdr lines)))
    (while rest
      (setq r (pack (list r "\n" (car rest))))
      (setq rest (cdr rest)))
    (if r r "")))

(de buf-from-string (str)
  (let ((lines NIL) (cur "") (chars (unpack str)))
    (while chars
      (if (equal (car chars) "\n")
        (prog (setq lines (append lines (list cur))) (setq cur ""))
        (setq cur (pack (list cur (car chars)))))
      (setq chars (cdr chars)))
    (buf-from-lines (append lines (list cur)))))
```

```lisp
# lib/vi/state.l — editor state as property list

(de s-get (state key)   (cadr (member key state)))

(de s-set (state key val)
  (if (member key state)
    (let ((s (copy state)))
      (setcar (cdr (member key s)) val) s)
    (append state (list key val))))

(de make-state (buf)
  (list
    'buffer      buf
    'cursor      '(1 . 0)
    'scroll      '(1 . 0)
    'mode        'normal
    'visual      NIL
    'cmd-buf     ""
    'search      ""
    'search-dir  'forward
    'search-mode NIL
    'undo        NIL
    'redo        NIL
    'unnamed-reg ""
    'registers   (let loop ((v (pvec)) (i 0))
                   (if (= i 26) v (loop (pvec-push v "") (1+ i))))
    'marks       (let loop ((v (pvec)) (i 0))
                   (if (= i 26) v (loop (pvec-push v NIL) (1+ i))))
    'macros      (let loop ((v (pvec)) (i 0))
                   (if (= i 26) v (loop (pvec-push v NIL) (1+ i))))
    'macro-rec   NIL
    'macro-inject NIL
    'last-find   NIL
    'last-cmd    NIL
    'filename    NIL
    'dirty       NIL
    'msg         ""
    'running     T))

(de undo-save (state)
  (s-set state 'undo
    (cons (cons (s-get state 'buffer) (s-get state 'cursor))
          (s-get state 'undo))))

(de undo-pop (state)
  (let ((stack (s-get state 'undo)))
    (if (null stack)
      (s-set state 'msg "Already at oldest change")
      (let* ((snap (car stack))
             (s2 (s-set state  'undo   (cdr stack)))
             (s3 (s-set s2     'buffer (car snap)))
             (s4 (s-set s3     'cursor (cdr snap))))
        (s-set s4 'redo
          (cons (cons (s-get state 'buffer) (s-get state 'cursor))
                (s-get state 'redo)))))))

(de redo-pop (state)
  (let ((stack (s-get state 'redo)))
    (if (null stack)
      (s-set state 'msg "Already at newest change")
      (let* ((snap (car stack))
             (s2 (s-set state  'redo   (cdr stack)))
             (s3 (s-set s2     'buffer (car snap)))
             (s4 (s-set s3     'cursor (cdr snap))))
        (s-set s4 'undo
          (cons (cons (s-get state 'buffer) (s-get state 'cursor))
                (s-get state 'undo)))))))
```

```lisp
# lib/vi/cursor.l — cursor clamping, movement, viewport
# Requires: buffer.l, state.l

(de cursor-clamp (buf cur)
  (let* ((row (max 1 (min (buf-nlines buf) (car cur))))
         (len (buf-line-len buf row))
         (col (max 0 (min (if (> len 0) (1- len) 0) (cdr cur)))))
    (cons row col)))

(de cursor-clamp-insert (buf cur)
  (let* ((row (max 1 (min (buf-nlines buf) (car cur))))
         (len (buf-line-len buf row))
         (col (max 0 (min len (cdr cur)))))
    (cons row col)))

(de cur-move-h (buf cur delta)
  (cursor-clamp buf (cons (car cur) (+ (cdr cur) delta))))

(de cur-move-v (buf cur delta)
  (cursor-clamp buf (cons (+ (car cur) delta) (cdr cur))))

(de cur-first-nonblank (buf row)
  (let ((chars (unpack (buf-line buf row))) (col 0))
    (while (and chars (equal (car chars) " "))
      (setq col (1+ col)) (setq chars (cdr chars)))
    col))

(de viewport-ensure (buf cur scroll cols rows)
  # rows = text rows (screen height - 1 for status bar)
  (let* ((row   (car cur))   (col   (cdr cur))
         (top   (car scroll)) (left  (cdr scroll))
         (bot   (+ top rows -1))
         (top2  (cond ((< row top)      row)
                      ((> row bot)      (- row rows -1))
                      (T                top)))
         (left2 (cond ((< col left)             (max 0 (- col 4)))
                      ((>= col (+ left cols))   (- col cols -5))
                      (T                         left))))
    (cons top2 left2)))
```

```lisp
# lib/vi/keys.l — key decoding and key-source coroutine
# Requires: lib/vi/term.l (term-read-byte, term-read-byte-timeout)

# (decode-csi-seq bytes) — pure: decode CSI from list of byte values (testable)
(de decode-csi-seq (bytes)
  (let ((params "") (b 0))
    (while (and bytes
                (let ((c (car bytes)))
                  (or (and (>= c 48) (<= c 57)) (= c 59))))
      (setq params (pack (list params (char (car bytes)))))
      (setq bytes (cdr bytes)))
    (setq b (if bytes (car bytes) 0))
    (cond
      ((= b 65) 'up)    ((= b 66) 'down)
      ((= b 67) 'right) ((= b 68) 'left)
      ((= b 72) 'home)  ((= b 70) 'end)
      ((and (= b 126) (equal params "5")) 'pgup)
      ((and (= b 126) (equal params "6")) 'pgdn)
      ((and (= b 126) (equal params "3")) 'del)
      ((and (= b 126) (equal params "2")) 'ins)
      (T 'unknown))))

(de decode-csi ()
  (let ((params "") (b (term-read-byte)))
    (while (or (and (>= b 48) (<= b 57)) (= b 59))
      (setq params (pack (list params (char b))))
      (setq b (term-read-byte)))
    (decode-csi-seq (list b))))

(de decode-ss3 ()
  (let ((b (term-read-byte)))
    (cond ((= b 65) 'up) ((= b 66) 'down) ((= b 67) 'right) ((= b 68) 'left)
          (T 'unknown))))

(de decode-esc ()
  (let ((b (term-read-byte-timeout 50)))
    (cond ((= b -1) 'esc) ((= b 91) (decode-csi)) ((= b 79) (decode-ss3)) (T 'esc))))

(de read-next-key ()
  (let ((b (term-read-byte)))
    (cond
      ((= b  27) (decode-esc))
      ((= b  13) 'ret)
      ((= b 127) 'bs)
      ((= b   9) 'tab)
      ((= b  18) 'ctrl-r)
      ((= b  22) 'ctrl-v)
      ((= b  23) 'ctrl-w)
      ((= b  21) 'ctrl-u)
      ((= b   4) 'ctrl-d)
      ((= b   6) 'ctrl-f)
      ((= b   2) 'ctrl-b)
      ((< b  32) NIL)
      (T (char b)))))

(de make-term-key-source ()
  (co (lambda (_)
    (while T (let ((k (read-next-key))) (when k (yield k)))))))
```

```lisp
# lib/vi/render.l — ANSI terminal renderer (pure L, no FFI)
# Requires: buffer.l, cursor.l, state.l

(de ansi-goto (row col)   (pack (list "\x1b[" (str row) ";" (str col) "H")))
(de ansi-clear-line ()    "\x1b[K")
(de ansi-cursor-hide ()   "\x1b[?25l")
(de ansi-cursor-show ()   "\x1b[?25h")
(de ansi-clear-screen ()  "\x1b[2J\x1b[H")
(de ansi-bold ()          "\x1b[1m")
(de ansi-reset ()         "\x1b[0m")
(de ansi-reverse ()       "\x1b[7m")

(de render-line (line left cols)
  # Return substring of line visible in viewport [left, left+cols)
  (let* ((chars (unpack line))
         (len   (length chars))
         (start (min left len))
         (end   (min (+ left cols) len))
         (vis   (let loop ((c chars) (i 0) (acc NIL))
                  (if (or (null c) (>= i end)) (reverse acc)
                    (loop (cdr c) (1+ i)
                      (if (>= i start) (cons (car c) acc) acc))))))
    (if vis (pack vis) "")))

(de render-status (state cols rows)
  (prin (ansi-goto rows 1))
  (prin (ansi-clear-line))
  (let ((mode (s-get state 'mode))
        (msg  (s-get state 'msg))
        (row  (car (s-get state 'cursor)))
        (col  (cdr (s-get state 'cursor))))
    (cond
      ((equal mode 'command)
       (prin (pack (list ":" (s-get state 'cmd-buf)))))
      ((equal mode 'insert)
       (prin (pack (list (ansi-bold) "-- INSERT --" (ansi-reset)))))
      ((member mode '(visual-char visual-line visual-block))
       (prin (pack (list (ansi-bold) "-- VISUAL --" (ansi-reset)))))
      (T (when (and msg (not (equal msg ""))) (prin msg))))
    (let ((pos (pack (list (str row) ":" (str (1+ col))))))
      (prin (pack (list (ansi-goto rows (max 1 (- cols (length (unpack pos))))) pos))))))

(de render-frame (state cols rows)
  (let* ((buf       (s-get state 'buffer))
         (cursor    (s-get state 'cursor))
         (scroll    (s-get state 'scroll))
         (top       (car scroll))
         (left      (cdr scroll))
         (crow      (car cursor))
         (ccol      (cdr cursor))
         (text-rows (1- rows)))
    (prin (ansi-cursor-hide))
    (let ((i 0))
      (while (< i text-rows)
        (let ((lnum (+ top i 1)))
          (prin (ansi-goto (1+ i) 1))
          (prin (ansi-clear-line))
          (if (> lnum (buf-nlines buf))
            (prin "~")
            (prin (render-line (buf-line buf lnum) left (1- cols)))))
        (setq i (1+ i))))
    (render-status state cols rows)
    (let ((scr-row (- crow top -1))
          (scr-col (- ccol left -1)))
      (when (and (>= scr-row 1) (<= scr-row text-rows) (>= scr-col 1))
        (prin (ansi-goto scr-row scr-col))))
    (prin (ansi-cursor-show))
    (flush)))

(de make-ansi-renderer (cols rows)
  (co (lambda (_)
    (while T (let ((state (yield NIL))) (render-frame state cols rows))))))
```

```lisp
(de word-end-forward (buf cur count)
  (let loop ((r (car cur)) (c (cdr cur)) (n count))
    (if (= n 0) (cons r c)

                       ((>= i len)

    ((equal key "b")  (word-start-backward buf cur count))
    ((equal key "e")  (word-end-forward    buf cur count))
    ((equal key "j")  (cursor-clamp buf (cons (+ (car cur) count) (cdr cur))))
    ((equal key "k")  (cursor-clamp buf (cons (- (car cur) count) (cdr cur))))
    ((equal key "$")  (cons (car cur) (max 0 (1- (buf-line-len buf (car cur))))))
    ((equal key "0")  (cons (car cur) 0))
    ((equal key "^")  (cons (car cur) (cur-first-nonblank buf (car cur))))

    ((equal key "H")  (cursor-clamp buf (cons 1 0)))

    ((equal key "L")  (cursor-clamp buf (cons (buf-nlines buf) 0)))

      ((equal key "G")
       (make-line-region row (buf-nlines buf)))
      (T (make-char-region row col row col)))))
```

```lisp
# Requires: state.l, cursor.l, buffer.l, ops.l
# Stubs at bottom replaced by later tasks (cmd.l, search.l, visual.l, macro.l)

(de enter-insert (state key)
  (let* ((buf (s-get state 'buffer))
         (cur (s-get state 'cursor))
         (row (car cur)) (col (cdr cur))
         (len (buf-line-len buf row))
         (st2 (s-set (undo-save state) 'mode 'insert)))
    (cond
      ((equal key "i") st2)
      ((equal key "I") (s-set st2 'cursor (cons row (cur-first-nonblank buf row))))
      ((equal key "a") (s-set st2 'cursor (cons row (min (1+ col) len))))
      ((equal key "A") (s-set st2 'cursor (cons row len)))
      ((equal key "o")
       (let* ((nb (buf-ins-line buf (1+ row) ""))
              (s3 (s-set st2 'buffer nb)))
         (s-set s3 'cursor (cons (1+ row) 0))))
      ((equal key "O")
       (let* ((nb (buf-ins-line buf row ""))
              (s3 (s-set st2 'buffer nb)))
         (s-set s3 'cursor (cons row 0))))
      (T st2))))

(de insert-key (state key)
  (let* ((buf (s-get state 'buffer))
         (cur (s-get state 'cursor))
         (row (car cur)) (col (cdr cur)))
    (cond
      ((equal key 'esc)
       (let* ((st2 (s-set state 'mode 'normal))
              (len (buf-line-len buf row))
              (nc  (max 0 (min (1- (max 1 len)) col))))
         (s-set st2 'cursor (cons row nc))))
      ((equal key 'ret)
       (let* ((nb  (buf-split-line buf row col))
              (st2 (s-set state 'buffer nb)))
         (s-set st2 'cursor (cons (1+ row) 0))))
      ((equal key 'bs)
       (if (> col 0)
         (let* ((nb  (buf-del-char buf row (1- col)))
                (st2 (s-set state 'buffer nb)))
           (s-set st2 'cursor (cons row (1- col))))
         (if (> row 1)
           (let* ((pl  (buf-line-len buf (1- row)))
                  (nb  (buf-join-lines buf (1- row)))
                  (st2 (s-set state 'buffer nb)))
             (s-set st2 'cursor (cons (1- row) pl)))
           state)))
      ((equal key 'del)
       (let ((len (buf-line-len buf row)))
         (if (< col len) (s-set state 'buffer (buf-del-char buf row col)) state)))
      ((string? key)
       (let* ((cp  (car (unpack key)))
              (nb  (buf-ins-char buf row col cp))
              (st2 (s-set state 'buffer nb)))
         (s-set st2 'cursor (cons row (1+ col)))))
      (T state))))

(de exec-prefix (state pfx key count)
  (cond
    ((equal pfx "g")
     (cond
       ((equal key "g")
        (s-set state 'cursor
          (cursor-clamp (s-get state 'buffer) (cons 1 (cdr (s-get state 'cursor))))))
       ((equal key "e")
        (s-set state 'cursor
          (word-end-forward (s-get state 'buffer) (s-get state 'cursor) count)))
       (T state)))
    ((equal pfx "Z")
     (cond
       ((equal key "Z") (cmd-write-quit state))
       ((equal key "Q") (s-set state 'running NIL))
       (T state)))
    ((equal pfx "r")   (replace-char state key))
    ((equal pfx "m")   (set-mark state key))
    ((equal pfx "'")   (jump-mark state key))
    ((member pfx '("f" "F" "t" "T")) (find-char state pfx key count))
    ((equal pfx "q")   (start-macro-record state key))
    ((equal pfx "@")   (play-macro state key count))
    (T state)))

(de make-mode-dispatch (init-state)
  (co (lambda (_)
    (let ((state init-state) (count 0) (op NIL) (pfx NIL))
      (while T
        (let ((key (yield state)))
          (when (and (s-get state 'msg) (not (equal (s-get state 'msg) "")))
            (setq state (s-set state 'msg "")))
          (cond
            (pfx
             (setq state (exec-prefix state pfx key (max 1 count)))
             (setq pfx NIL) (setq count 0) (setq op NIL))

            ((equal (s-get state 'mode) 'insert)
             (setq state (insert-key state key)))

            ((equal (s-get state 'mode) 'command)
             (setq state (cmd-key state key)))

            ((member (s-get state 'mode) '(visual-char visual-line visual-block))
             (setq state (visual-key state key count))
             (setq count 0))
```

```lisp
# lib/vi/search.l — pure L regex engine and vi search operations
# Supports: . * + ? ^ $ [...] [^...] \w \d \s
# regex-match returns (start . end) where end is exclusive, or NIL

# ---- Regex compiler ----
# Each compiled atom: (atom-spec quant)
# atom-spec: 'dot | 'bol | 'eol | (lit ch) | (cls chars) | (ncls chars)
#            | (cls-w) | (cls-d) | (cls-s)
# quant: NIL | '* | '+ | '?

(de re-compile (pat)
  (let ((chars (unpack pat)) (atoms NIL))
    (while chars
      (let* ((c (car chars))
             (atom
               (cond
                 ((equal c "^")  (setq chars (cdr chars)) 'bol)
                 ((equal c "$")  (setq chars (cdr chars)) 'eol)
                 ((equal c ".")  (setq chars (cdr chars)) 'dot)
                 ((equal c "\\")
                  (let ((nc (cadr chars)))
                    (setq chars (cddr chars))
                    (cond
                      ((equal nc "w") '(cls-w))
                      ((equal nc "d") '(cls-d))
                      ((equal nc "s") '(cls-s))
                      (T              (list 'lit nc)))))
                 ((equal c "[")
                  (setq chars (cdr chars))
                  (let ((neg (equal (car chars) "^")) (cset NIL))
                    (when neg (setq chars (cdr chars)))
                    (while (and chars (not (equal (car chars) "]")))
                      (if (and (cdr chars) (equal (cadr chars) "-") (cddr chars)
                               (not (equal (caddr chars) "]")))
                        (let ((lo (car (unpack (car chars))))
                              (hi (car (unpack (caddr chars)))))
                          (let lp ((i lo))
                            (when (<= i hi) (setq cset (cons (char i) cset)) (lp (1+ i))))
                          (setq chars (cdddr chars)))
                        (progn (setq cset (cons (car chars) cset)) (setq chars (cdr chars)))))
                    (setq chars (cdr chars))   ; consume ']'
                    (if neg (list 'ncls cset) (list 'cls cset))))
                 (T (setq chars (cdr chars)) (list 'lit c))))
             (quant
               (cond
                 ((and chars (equal (car chars) "*")) (setq chars (cdr chars)) '*)
                 ((and chars (equal (car chars) "+")) (setq chars (cdr chars)) '+)
                 ((and chars (equal (car chars) "?")) (setq chars (cdr chars)) '?)
                 (T NIL))))
        (when atom (setq atoms (append atoms (list (list atom quant)))))))
    atoms))

(de re-match-char (atom ch)
  (cond
    ((equal atom 'dot)     (not (equal ch "\n")))
    ((equal atom 'bol)     NIL)
    ((equal atom 'eol)     NIL)
    ((and (pair? atom) (equal (car atom) 'lit))  (equal ch (cadr atom)))
    ((and (pair? atom) (equal (car atom) 'cls))  (member ch (cadr atom)))
    ((and (pair? atom) (equal (car atom) 'ncls)) (not (member ch (cadr atom))))
    ((and (pair? atom) (equal (car atom) 'cls-w))
     (let ((c (car (unpack ch))))
       (or (and (>= c 65) (<= c 90)) (and (>= c 97) (<= c 122))
           (and (>= c 48) (<= c 57)) (= c 95))))
    ((and (pair? atom) (equal (car atom) 'cls-d))
     (let ((c (car (unpack ch)))) (and (>= c 48) (<= c 57))))
    ((and (pair? atom) (equal (car atom) 'cls-s))
     (or (equal ch " ") (equal ch "\t") (equal ch "\n")))
    (T NIL)))

(de re-nth (chars i)
  (if (or (null chars) (< i 0)) NIL
    (if (= i 0) (car chars) (re-nth (cdr chars) (1- i)))))

(de re-try (atoms chars start pos)
  (cond
    ((null atoms) pos)
    (T
     (let* ((entry (car atoms))
            (atom  (car entry))
            (quant (cadr entry))
            (rest  (cdr atoms))
            (ch    (re-nth chars pos)))
       (cond
         ((equal atom 'bol)
          (if (= pos start) (re-try rest chars start pos) NIL))
         ((equal atom 'eol)
          (if (null ch) (re-try rest chars start pos) NIL))
         ((equal quant '*)
          (let ((ms (let lp ((i pos) (acc (list pos)))
                      (let ((c (re-nth chars i)))
                        (if (and c (re-match-char atom c))
                          (lp (1+ i) (cons (1+ i) acc))
                          acc)))))
            (let try ((ls ms))
              (if (null ls) NIL
                (or (re-try rest chars start (car ls)) (try (cdr ls)))))))
         ((equal quant '+)
          (if (and ch (re-match-char atom ch))
            (let ((ms (let lp ((i (1+ pos)) (acc (list (1+ pos))))
                        (let ((c (re-nth chars i)))
                          (if (and c (re-match-char atom c))
                            (lp (1+ i) (cons (1+ i) acc))
                            acc)))))
              (let try ((ls ms))
                (if (null ls) NIL
                  (or (re-try rest chars start (car ls)) (try (cdr ls))))))
            NIL))
         ((equal quant '?)
          (or (and ch (re-match-char atom ch) (re-try rest chars start (1+ pos)))
              (re-try rest chars start pos)))
         (T
          (if (and ch (re-match-char atom ch))
            (re-try rest chars start (1+ pos))
            NIL)))))))

(de regex-match (pattern str start)
  (let* ((atoms    (re-compile pattern))
         (chars    (unpack str))
         (len      (length chars))
         (anchored (and atoms (equal (car (car atoms)) 'bol))))
    (if anchored
      (let ((end (re-try atoms chars start start)))
        (if end (cons start end) NIL))
      (let loop ((i start))
        (if (> i len) NIL
          (let ((end (re-try atoms chars i i)))
            (if end (cons i end) (loop (1+ i)))))))))

# ---- Search operations ----

(de search-forward (buf pat from-row from-col)
  (let ((nlines (buf-nlines buf)))
    (let loop ((r from-row) (first T))
      (if (and (> r nlines) (not first)) NIL
        (let* ((rr   (if (> r nlines) 1 r))
               (line (buf-line buf rr))
               (sc   (if (and first (= rr from-row)) (1+ from-col) 0))
               (m    (regex-match pat line sc)))
          (if m (cons rr (car m))
            (loop (if (= rr nlines) 1 (1+ rr))
              (if (= rr nlines) (not first) NIL))))))))

(de search-backward (buf pat from-row from-col)
  (let ((nlines (buf-nlines buf)))
    (let loop ((r from-row) (first T))
      (if (and (< r 1) (not first)) NIL
        (let* ((rr   (if (< r 1) nlines r))
               (line (buf-line buf rr))
               (ec   (if (and first (= rr from-row)) from-col (length (unpack line))))
               (m2   (let lp ((i 0) (last NIL))
                       (if (> i ec) last
                         (let ((mm (regex-match pat line i)))
                           (if (and mm (<= (car mm) ec))
                             (lp (1+ (car mm)) mm)
                             (lp (1+ i) last)))))))
          (if m2 (cons rr (car m2))
            (loop (if (= rr 1) nlines (1- rr))
              (if (= rr 1) (not first) NIL))))))))

(de start-search (state dir)
  (s-set (s-set (s-set (s-set state 'mode 'command) 'cmd-buf "")
                       'search-dir dir)
         'search-mode T))

(de search-exec-key (state key)
  (let ((buf (s-get state 'cmd-buf)))
    (cond
      ((equal key 'esc)
       (s-set (s-set (s-set state 'mode 'normal) 'cmd-buf "") 'search-mode NIL))
      ((equal key 'ret)
       (let* ((pat (s-get state 'cmd-buf))
              (st2 (s-set (s-set (s-set state 'mode 'normal) 'search pat) 'search-mode NIL))
              (st3 (search-jump st2)))
         (s-set st3 'cmd-buf "")))
      ((equal key 'bs)
       (let ((len (length (unpack buf))))
         (if (> len 0)
           (s-set state 'cmd-buf (sub buf 0 (1- len)))
           (s-set (s-set (s-set state 'mode 'normal) 'cmd-buf "") 'search-mode NIL))))
      ((string? key)
       (s-set state 'cmd-buf (pack (list buf key))))
      (T state))))

(de search-jump (state)
  (let* ((buf (s-get state 'buffer))
         (cur (s-get state 'cursor))
         (pat (s-get state 'search))
         (dir (s-get state 'search-dir))
         (row (car cur)) (col (cdr cur))
         (hit (if (equal dir 'forward)
                (search-forward  buf pat row col)
                (search-backward buf pat row col))))
    (if hit
      (s-set state 'cursor (cons (car hit) (cdr hit)))
      (s-set state 'msg (pack (list "Pattern not found: " pat))))))

(de search-next (state count)
  (let loop ((st state) (n (max 1 count)))
    (if (= n 0) st (loop (search-jump st) (1- n)))))

(de search-prev (state count)
  (let* ((dir (s-get state 'search-dir))
         (fl  (s-set state 'search-dir (if (equal dir 'forward) 'backward 'forward)))
         (res (search-next fl count)))
    (s-set res 'search-dir dir)))

  # arg is "from/to/flags" (the part after :s)
  (let* ((parts  (split arg "/"))

             (let ((m (regex-match from line 0)))
               (if m (pack (list (sub line 0 (car m)) to (sub line (cdr m)))) line))))
         (nb   (buf-set-line buf row new-line))
         (st2  (s-set state 'buffer nb)))
    (s-set st2 'dirty T)))
```

```lisp
         (ar (car anchor)) (ac (cdr anchor))
         (cr (car cur))    (cc (cdr cur))
         (r1 (min ar cr))  (r2 (max ar cr))
         (c1 (cond ((< ar cr) ac) ((> ar cr) cc) (T (min ac cc))))
         (c2 (cond ((< ar cr) cc) ((> ar cr) ac) (T (max ac cc)))))
    (if (equal mode 'visual-line)
      (make-line-region r1 r2)
      (make-char-region r1 c1 r2 c2))))

(de visual-key (state key count)
  (cond
    ((equal key 'esc)
     (s-set (s-set state 'mode 'normal) 'visual NIL))
    ((member key '("d" "x"))
     (s-set (s-set (delete-region (undo-save state) (visual-region state) T) 'mode 'normal) 'visual NIL))
    ((equal key "y")
     (s-set (s-set (yank-region state (visual-region state)) 'mode 'normal) 'visual NIL))
    ((equal key "c")
     (s-set (s-set (delete-region (undo-save state) (visual-region state) T) 'mode 'insert) 'visual NIL))
    ((equal key ">")
     (s-set (s-set (indent-region (undo-save state) (visual-region state) 1) 'mode 'normal) 'visual NIL))
    ((equal key "<")
     (s-set (s-set (indent-region (undo-save state) (visual-region state) -1) 'mode 'normal) 'visual NIL))
    ; motion keys extend selection
    ((or (member key '("h" "l" "j" "k" "w" "b" "e" "0" "^" "$" "G"))
         (motion-key? key))
     (exec-motion state key (max 1 count) NIL))
    (T state)))
```

```lisp
# lib/vi/macro.l — macro recording (qa-qz) and playback (@a-@z)
# Requires: state.l

(de start-macro-record (state key)
  (if (and (string? key)
           (>= (car (unpack key)) 97) (<= (car (unpack key)) 122))
    (let* ((idx     (reg-idx key))
           (macros  (pvec-put (s-get state 'macros) idx NIL))
           (st2     (s-set state 'macros macros))
           (st3     (s-set st2 'macro-rec idx)))
      (s-set st3 'msg (pack (list "recording @" key))))
    state))

(de stop-macro-record (state)
  (s-set (s-set state 'macro-rec NIL) 'msg "Macro recorded"))

(de macro-add-key (state key)
  (let* ((idx    (s-get state 'macro-rec))
         (macros (s-get state 'macros))
         (cur    (or (pvec-get macros idx) NIL))
         (new    (append cur (list key))))
    (s-set state 'macros (pvec-put macros idx new))))

(de play-macro (state key count)
  (if (and (string? key)
           (>= (car (unpack key)) 97) (<= (car (unpack key)) 122))
    (let* ((idx  (reg-idx key))
           (keys (or (pvec-get (s-get state 'macros) idx) NIL))
           (inject (let loop ((n count) (acc NIL))
                     (if (= n 0) acc (loop (1- n) (append acc keys))))))
      (s-set state 'macro-inject inject))
    state))

(de make-macro-filter (key-src state-cell)
  # state-cell is a 1-element list holding current state (mutable ref)
  # The filter checks (car state-cell) for macro-inject on each cycle.
  (co (lambda (_)
    (let ((buf NIL))
      (while T
        (let ((inject (s-get (car state-cell) 'macro-inject)))
          (when inject
            (setq buf (append buf inject))
            (setcar state-cell (s-set (car state-cell) 'macro-inject NIL))))
        (let ((key (if buf
                     (let ((k (car buf)))
                       (setq buf (cdr buf)) k)
                     (co-resume key-src NIL))))
          ; record key if macro recording active
          (let ((idx (s-get (car state-cell) 'macro-rec)))
            (when (and idx (not (equal key "q")))
              (setcar state-cell (macro-add-key (car state-cell) key))))
          (yield key)))))))
```

```lisp
# lib/vi/gfx.l — SDL graphical mode key source and renderer
# Requires: HAVE_SDL build, sdl-wait-event, sdl-text, sdl-font-load

(de sdl-key->vi-key (keyname mod)
  (let ((shift (and mod (not (= (bitand mod 1) 0)))))
    (cond
      ((equal keyname "Return")    'ret)
      ((equal keyname "Backspace") 'bs)
      ((equal keyname "Escape")    'esc)
      ((equal keyname "Tab")       'tab)
      ((equal keyname "Up")        'up)
      ((equal keyname "Down")      'down)
      ((equal keyname "Left")      'left)
      ((equal keyname "Right")     'right)
      ((equal keyname "Home")      'home)
      ((equal keyname "End")       'end)
      ((equal keyname "Page_Up")   'pgup)
      ((equal keyname "Page_Down") 'pgdn)
      ((equal keyname "Delete")    'del)
      ((= (length (unpack keyname)) 1)
       (if shift (upcase keyname) (downcase keyname)))
      (T NIL))))

(de make-sdl-key-source ()
  (co (lambda (_)
    (while T
      (let* ((ev  (sdl-wait-event))
             (tp  (cadr (member "type" ev)))
             (key (when (equal tp "keydown")
                    (sdl-key->vi-key (cadr (member "key" ev))
                                     (cadr (member "mod" ev))))))
        (when key (yield key)))))))

(de make-sdl-renderer (cell-w cell-h scols srows)
  (co (lambda (_)
    (while T
      (let ((state (yield NIL)))
        (sdl-color 200 200 200)
        (sdl-clear)
        (let* ((buf    (s-get state 'buffer))
               (cursor (s-get state 'cursor))
               (scroll (s-get state 'scroll))
               (top    (car scroll)) (left (cdr scroll))
               (crow   (car cursor)) (ccol (cdr cursor))
               (mode   (s-get state 'mode))
               (text-rows (1- srows)))
          (let ((i 0))
            (while (< i text-rows)
              (let* ((lnum (+ top i 1))
                     (y    (* i cell-h)))
                (if (> lnum (buf-nlines buf))
                  (progn (sdl-color 100 100 100) (sdl-text 0 y "~") (sdl-color 200 200 200))
                  (let* ((line (buf-line buf lnum))
                         (chars (unpack line))
                         (vis   (let loop ((c chars) (i2 0) (acc NIL))
                                  (if (or (null c) (>= i2 (+ left scols))) (reverse acc)
                                    (loop (cdr c) (1+ i2)
                                      (if (>= i2 left) (cons (car c) acc) acc))))))
                    (sdl-text 0 y (if vis (pack vis) "")))))
              (setq i (1+ i))))
          ; cursor
          (let* ((scr-row (- crow top))
                 (scr-col (- ccol left))
                 (cx (* scr-col cell-w))
                 (cy (* scr-row cell-h)))
            (when (and (>= scr-row 0) (< scr-row text-rows))
              (sdl-color 200 200 50)
              (sdl-rect cx cy cell-w cell-h)))
          ; status bar
          (let ((sy (* text-rows cell-h)))
            (sdl-color 50 50 50)
            (sdl-fill 0 sy (* scols cell-w) cell-h)
            (sdl-color 200 200 200)
            (sdl-text 0 sy
              (cond
                ((equal mode 'insert)  "-- INSERT --")
                ((equal mode 'command) (pack (list ":" (s-get state 'cmd-buf))))
                ((member mode '(visual-char visual-line visual-block)) "-- VISUAL --")
                (T (or (s-get state 'filename) "[No Name]")))))
          (sdl-present)))))))

(de vi-gfx-start (filename)
  (let* ((cell-w 8) (cell-h 16) (scols 100) (srows 40))
    (sdl-window "vi" (* scols cell-w) (* srows cell-h))
    (sdl-font-load "assets/monospace.ttf" 14)
    (let* ((buf    (if filename (or (file-read filename) (buf-empty)) (buf-empty)))
           (st0    (make-state buf))
           (st0    (if filename (s-set st0 'filename filename) st0))
           (sc     (list st0))
           (ksrc   (make-sdl-key-source))
           (mac    (make-macro-filter ksrc sc))
           (rend   (make-sdl-renderer cell-w cell-h scols srows))
           (disp   (make-mode-dispatch st0))
           (state  (co-resume disp NIL)))
      (setcar sc state)
      (co-resume rend state)
      (while (s-get state 'running)
        (let* ((key (co-resume mac NIL))
               (key (if (and (equal key "q") (s-get state 'macro-rec))
                      (progn (setq state (stop-macro-record state)) (setcar sc state) NIL)
                      key))
               (st2 (if key (co-resume disp key) state)))
          (setq state st2)
          (setcar sc state)
          (co-resume rend state)))
      (sdl-quit))))
```
